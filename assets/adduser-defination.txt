action: addUser
    protocol: http
    url: http://apis.usermanagement.org/users/
    method: POST
    format: application/json
    properties:

            required: true
            type: string
            validation: alphanumeric

            required: true
            type: string
            validation: alphanumeric
            min-size: 8
            max-size: 64
    returns:
        format: application/json
        url: http://apis.usermanagement.org/users/{userId}
